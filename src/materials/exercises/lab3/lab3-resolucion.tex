% ---- Task 2 -----------------------------------------------------------------
\section{Task 2 --- Privilege Escalation (4 points)}

\textbf{Command used:} \texttt{sudo /usr/bin/find}\\
\textbf{Misconfiguration:} \texttt{/etc/sudoers.d/devops} contains \texttt{NOPASSWD: /usr/bin/find}\\
\textbf{MITRE technique:} T1548.003 --- Abuse Elevation Control Mechanism: Sudo and Sudo Caching\\
\textbf{Prevention:} Remove \texttt{NOPASSWD}; apply principle of least privilege; use \texttt{visudo}

Key log lines:
\begin{lstlisting}
Apr 24 03:18:01 helix-web sudo: devops : TTY=pts/0 ;
  PWD=/home/devops ; USER=root ; COMMAND=/usr/bin/find
\end{lstlisting}

\begin{lstlisting}[language=bash]
# /etc/sudoers.d/devops
devops ALL=(ALL) NOPASSWD: /usr/bin/find
\end{lstlisting}

\begin{mdframed}[backgroundcolor=labgray, linewidth=0.5pt]
\textbf{GTFOBins vector:} \texttt{find}'s \texttt{-exec} flag executes arbitrary commands as root.\\
\texttt{sudo find / -name "*.conf" -exec cat \{\} \textbackslash;}
\end{mdframed}

\section{Task 3 --- Persistence \& Post-Exploitation (3 points)}

\textbf{Backdoor account:} \texttt{maint} (UID 1003)\\
\textbf{Privileges:} \texttt{NOPASSWD: ALL} via \texttt{/etc/sudoers.d/maint}

Key log lines:
\begin{lstlisting}
Apr 24 03:19:15 helix-web useradd[2041]: new user: name=maint,
  UID=1003, GID=1003, home=/home/maint, shell=/bin/bash
\end{lstlisting}

Full attacker command history (\texttt{/home/devops/.bash\_history}):
\begin{lstlisting}[language=bash]
id
whoami
sudo /usr/bin/find / -name "*.conf" -exec cat {} \;
useradd -m maint
echo "maint ALL=(ALL) NOPASSWD: ALL" > /etc/sudoers.d/maint
chmod 440 /etc/sudoers.d/maint
exit
\end{lstlisting}

\section{Task 6 --- Incident Report (4 points)}

\subsection*{Indicators of Compromise (IOCs)}
\begin{itemize}[leftmargin=1.5em]
  \item \textbf{IP:} \texttt{185.220.101.47} (external attacker)
  \item \textbf{User:} \texttt{devops} (compromised via brute force)
  \item \textbf{Backdoor:} \texttt{maint} account (UID~1003, \texttt{NOPASSWD: ALL})
  \item \textbf{Files:} \texttt{/opt/helix/data/clients.db}, \texttt{.exfil\_marker}
  \item \textbf{Technique:} GTFOBins \texttt{sudo find} (T1548.003)
\end{itemize}

\subsection*{Complete attack timeline}
\begin{table}[H]

\begin{tabularx}{\textwidth}{llX}

\subsection*{Recommended remediation}
\begin{enumerate}[leftmargin=2em]
  \item Disable \texttt{devops} and \texttt{maint} accounts immediately
  \item Remove \texttt{NOPASSWD} from \texttt{/etc/sudoers.d/devops}
  \item Rotate all credentials on both servers
  \item Implement SSH key-based auth; disable password authentication
  \item Deploy \texttt{fail2ban} + \texttt{auditd} on all servers
  \item Apply least-privilege ownership on \texttt{/opt/helix/data/}
\end{enumerate}
